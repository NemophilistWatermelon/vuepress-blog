# Props (TypeScript 写法)

# 有默认值

// 有默认值传递写法 
interface Props {
  msg?: string,
  content?: string
}

const props = withDefaults(defineProps<Props>(), {
  msg: 'hello',
  content: '默认文本'
})

# 无默认值


defineProps<{
   msg: string,
   content: string
}>()

# Props (JavaScript 写法)

# 有默认值

// 有默认值传递写法 
defineProps({
   msg: String,
   content: {
		type: String,

# Emit 事件 (TypeScript 写法)

# 有默认值

const emit = defineEmits<{
  (e: ‘click’, num:number): void,
  (e: ‘wock’, num:number): void
}>()

const clickHere = (_:any):void => {
  emit(‘wock’, 12112)
}


	子组件
	<button @click='clickHere'/>

	父组件
	<Child @wock='xxxEventName'>
